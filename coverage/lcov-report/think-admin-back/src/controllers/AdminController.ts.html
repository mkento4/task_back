import { Request, Response } from 'express';
import * as jwt from 'jsonwebtoken';
import { getRepository } from 'typeorm';
import { validate } from 'class-validator';
 
import { Admins } from '../entity/Admin';
import config from '../config/config';
import toHash from '../utils/to_hash';
 
class AdminController {
  static login = async (req: Request, res: Response): Promise<void> => {
    //Check if username and password are set
    const { userName, password } = req.body;
    if (!(userName && password)) {
      res.status(400).send();
    }
 
    //Get user from database
    const adminRepository = getRepository(Admins);
    let admins!: Admins | undefined;
    try {
      admins = await adminRepository
        .createQueryBuilder('Admins')
        .addSelect('Admins.password')
        .where({ userName: userName })
        .getOne();
    } catch (error) {
      res.status(401).send(`error: ${error}`);
    }
 
    if (admins === undefined) {
      res.status(404).send();
      return;
    }
 
    //Check if encrypted password match
    if (!admins.checkIfUnencryptedPasswordIsValid(password)) {
      res.status(401).send();
      return;
    }
 
    //Sing JWT, valid for 1 hour
    const token = jwt.sign(
      { id: admins.id, userName: admins.userName },
      config.jwtSecret,
      { expiresIn: '1h' }
    );
 
    // set token
    res.setHeader('authorization', token);
 
    //Send the jwt in the response
    res.send({ token: token });
  };
 
  static changePassword = async (
    req: Request,
    res: Response
  ): Promise<void> => {
    //Get ID from JWT
    const id = res.locals.jwtPayload.id;
 
    //Get parameters from the body
    const { oldPassword, newPassword } = req.body;
    if (!(oldPassword && newPassword)) {
      res.status(400).send();
    }
 
    //Get user from the database
    const adminRepository = getRepository(Admins);
    let admins!: Admins | undefined;
    try {
      admins = await adminRepository
        .createQueryBuilder('Admin')
        .addSelect('Admin.password')
        .where(id)
        .getOne();
    } catch (id) {
      res.status(401).send();
    }
    if (admins === undefined) {
      res.status(404).send();
      return;
    }
 
    //Check if old password matchs
    if (!admins.checkIfUnencryptedPasswordIsValid(oldPassword)) {
      res.status(401).send();
      return;
    }
 
    //Validate de model (password length)
    admins.password = newPassword;
    const errors = await validate(Admins);
    if (errors.length > 0) {
      res.status(400).send(errors);
      return;
    }
    //Hash the new password and save
    admins.password = toHash(admins.password);
    await adminRepository.save(admins);
 
    res.status(204).send();
  };
 
  static me = async (req: Request, res: Response): Promise<void> => {
    //Get the jwt token from the head
    const bearerTokens = (req.headers['authorization'] as string).split(' ');
    if (bearerTokens[0].toLowerCase() !== 'bearer') {
      //If token is not valid, respond with 401 (unauthorized)
      res.status(401).send('Invalid token2');
      return;
    }
 
    let jwtPayload;
 
    const token = bearerTokens[1];
 
    try {
      //Decode token
      jwtPayload = jwt.verify(token, config.jwtSecret) as any;
 
      //Get user from id
      const adminRepository = getRepository(Admins);
      const admins = await adminRepository.findOneOrFail(jwtPayload.id);
      res.send({ admins: admins });
    } catch (e) {
      res.status(401).send();
      return;
    }
  };
}
 
export default AdminController;
